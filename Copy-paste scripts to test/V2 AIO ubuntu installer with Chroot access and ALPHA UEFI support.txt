echo "======================================================="

cat << 'EOF_3' > 3_firmware_fetcher.sh
#!/bin/bash
set -e
source build.env

echo "======================================================="
echo "   [3/6] Firmware Fetcher (Mobian Extraction)"
echo "======================================================="

if [ -d "$FIRMWARE_STASH" ] && [ "$(ls -A $FIRMWARE_STASH 2>/dev/null)" ]; then
    echo ">>> Found existing firmware stash at $FIRMWARE_STASH. Skipping network fetch."
else
    echo ">>> Local firmware stash is empty."
    echo ">>> Please ensure your Poco F1 is powered on, connected to Wi-Fi,"
    echo ">>> and actively running Mobian."
    echo ""
    read -p "Enter Poco F1 IP Address (e.g., 192.168.1.50): " PHONE_IP
    read -p "Enter Mobian username [default: mobian]: " PHONE_USER
    PHONE_USER=${PHONE_USER:-mobian}
    read -s -p "Enter Mobian password: " PHONE_PASS
    echo ""
    
    echo ">>> [Phone Side] Archiving hardware profiles via SSH..."
    sshpass -p "$PHONE_PASS" ssh -o StrictHostKeyChecking=no "$PHONE_USER@$PHONE_IP" \
        "echo '$PHONE_PASS' | sudo -S tar -czpf ~/mobian_harvest.tar.gz /usr/share/alsa/ucm2/ /etc/ModemManager/ /lib/udev/rules.d/"
    
    echo ">>> [Host Side] Downloading the archive..."
    mkdir -p "$FIRMWARE_STASH"
    sshpass -p "$PHONE_PASS" scp -o StrictHostKeyChecking=no "$PHONE_USER@$PHONE_IP:~/mobian_harvest.tar.gz" "$FIRMWARE_STASH/raw_harvest.tar.gz"
    
    echo ">>> [Host Side] Extracting into stash..."
    tar -xzpf "$FIRMWARE_STASH/raw_harvest.tar.gz" -C "$FIRMWARE_STASH/"
    rm "$FIRMWARE_STASH/raw_harvest.tar.gz"
    
    echo ">>> [Phone Side] Cleaning up temporary files..."
    sshpass -p "$PHONE_PASS" ssh -o StrictHostKeyChecking=no "$PHONE_USER@$PHONE_IP" "rm ~/mobian_harvest.tar.gz"
fi

echo ">>> [Host Side] Zipping firmware stash for later upload/distribution..."
if [ ! -f "$HOME/beryllium_mobian_firmware.zip" ]; then
    cd "$FIRMWARE_STASH" && zip -r "$HOME/beryllium_mobian_firmware.zip" ./* && cd - > /dev/null
    echo ">>> Portable zip backup created at $HOME/beryllium_mobian_firmware.zip"
else
    echo ">>> Zip backup already exists at $HOME/beryllium_mobian_firmware.zip"
fi

echo ">>> Firmware successfully stashed and secured."
echo ">>> Proceed to Script 4 (The Transplant)."
EOF_3

cat << 'EOF_4' > 4_the_transplant.sh
#!/bin/bash
set -e
source build.env

echo "======================================================="
echo "   [4/6] The Transplant (Base Build & UEFI Prep)"
echo "======================================================="
echo ">>> Target: Beryllium | User: $USERNAME | UI: $UI_NAME"
echo ">>> Image Size: ${IMAGE_SIZE}GB"
echo ">>> ---------------------------------------------------"

# --- Debootstrap Phase ---
if [ ! -d "ubuntu_rootfs" ]; then
    echo ">>> [Debootstrap] Building Ubuntu $UBUNTU_RELEASE (arm64)..."
    sudo debootstrap --arch=arm64 --foreign "$UBUNTU_RELEASE" ubuntu_rootfs http://ports.ubuntu.com/
    sudo cp /usr/bin/qemu-aarch64-static ubuntu_rootfs/usr/bin/
    sudo chroot ubuntu_rootfs /debootstrap/debootstrap --second-stage
else
    echo ">>> [Debootstrap] ubuntu_rootfs already exists, skipping base generation."
fi

# --- Injection Phase ---
echo ">>> [Merge] Injecting pmOS kernels & firmware..."
sudo cp -a pmos_harvest/lib/modules/. ubuntu_rootfs/lib/modules/
sudo cp -a pmos_harvest/lib/firmware/. ubuntu_rootfs/lib/firmware/

# --- UEFI Boot Payload Extraction ---
echo ">>> [Extract] Pulling raw kernel (vmlinuz), initramfs, and DTBs for UEFI/EDK2..."
mkdir -p uefi_boot_payload
if ls pmos_harvest/boot/vmlinuz* 1> /dev/null 2>&1; then
    sudo cp -L pmos_harvest/boot/vmlinuz* uefi_boot_payload/
    sudo cp -L pmos_harvest/boot/initramfs* uefi_boot_payload/
    
    if [ -d "pmos_harvest/boot/dtbs" ]; then
        sudo cp -Lr pmos_harvest/boot/dtbs uefi_boot_payload/
    fi
    sudo chown -R $(whoami):$(whoami) uefi_boot_payload/
    echo ">>> Raw UEFI boot files successfully extracted."
else
    echo ">>> [WARNING] No vmlinuz found in pmos_harvest/boot/. Could not extract raw files."
fi

echo ">>> [Merge] Injecting stashed Mobian hardware profiles..."
if [ -d "$FIRMWARE_STASH/usr/share/alsa/ucm2" ]; then
    sudo mkdir -p ubuntu_rootfs/usr/share/alsa/ucm2/
    sudo cp -a "$FIRMWARE_STASH/usr/share/alsa/ucm2/." ubuntu_rootfs/usr/share/alsa/ucm2/ || true
fi
if [ -d "$FIRMWARE_STASH/etc/ModemManager" ]; then
    sudo mkdir -p ubuntu_rootfs/etc/ModemManager/
    sudo cp -a "$FIRMWARE_STASH/etc/ModemManager/." ubuntu_rootfs/etc/ModemManager/ || true
fi
if [ -d "$FIRMWARE_STASH/lib/udev/rules.d" ]; then
    sudo mkdir -p ubuntu_rootfs/lib/udev/rules.d/
    sudo cp -a "$FIRMWARE_STASH/lib/udev/rules.d/." ubuntu_rootfs/lib/udev/rules.d/ || true
fi

# --- Setup Phase ---
echo ">>> [Config] Expanding repositories, installing UI, user, and GRUB EFI..."
sudo chroot ubuntu_rootfs /bin/bash << CHROOT_EOF
echo "nameserver 8.8.8.8" > /etc/resolv.conf

cat << APT_EOF > /etc/apt/sources.list
deb http://ports.ubuntu.com/ubuntu-ports/ $UBUNTU_RELEASE main restricted universe multiverse
deb http://ports.ubuntu.com/ubuntu-ports/ $UBUNTU_RELEASE-updates main restricted universe multiverse
deb http://ports.ubuntu.com/ubuntu-ports/ $UBUNTU_RELEASE-security main restricted universe multiverse
deb http://ports.ubuntu.com/ubuntu-ports/ $UBUNTU_RELEASE-backports main restricted universe multiverse
APT_EOF

# Only create user if it doesn't exist
if ! id "$USERNAME" &>/dev/null; then
    useradd -m -s /bin/bash $USERNAME
    echo "$USERNAME:$PASSWORD" | chpasswd
    usermod -aG sudo,video,audio,plugdev $USERNAME
fi

apt-get update && apt-get upgrade -y
export DEBIAN_FRONTEND=noninteractive
apt-get install -y $UI_PKG $EXTRA_PKG modemmanager network-manager systemd-resolved grub-efi-arm64-bin
echo "/usr/sbin/lightdm" > /etc/X11/default-display-manager 2>/dev/null || true
dpkg-reconfigure lightdm 2>/dev/null || true

# Generate monolithic GRUB EFI binary
grub-mkimage -o /BOOTAA64.EFI -O arm64-efi -p /EFI/BOOT \
    fat ext2 part_gpt normal linux all_video boot configfile search search_fs_file search_fs_uuid search_label minicmd
CHROOT_EOF

# --- ESP Staging Phase ---
echo ">>> [ESP Prep] Checking for UEFI payload files..."
KERNEL_FILE=$(ls uefi_boot_payload/vmlinuz* 2>/dev/null | head -n 1 || true)
INITRD_FILE=$(ls uefi_boot_payload/initramfs* 2>/dev/null | head -n 1 || true)

if [ -z "$KERNEL_FILE" ] || [ -z "$INITRD_FILE" ]; then
    echo ">>> [ERROR] Missing vmlinuz or initramfs in uefi_boot_payload/!"
    echo ">>> Check your pmOS generation. Halting to prevent broken bootloader."
    exit 1
fi

KNAME=$(basename "$KERNEL_FILE")
INAME=$(basename "$INITRD_FILE")

echo ">>> [ESP Prep] Assembling unsealed ESP folder..."
mkdir -p esp_folder/EFI/BOOT
sudo mv ubuntu_rootfs/BOOTAA64.EFI esp_folder/EFI/BOOT/
cp "$KERNEL_FILE" esp_folder/
cp "$INITRD_FILE" esp_folder/
if [ -d "uefi_boot_payload/dtbs" ]; then
    cp -r uefi_boot_payload/dtbs esp_folder/
fi

cat << GRUB_EOF > esp_folder/EFI/BOOT/grub.cfg
set timeout=3
menuentry "Ubuntu (Beryllium UEFI)" {
    linux /$KNAME root=/dev/sda2 rw rootwait console=tty0
    initrd /$INAME
    # WARNING: Adjust the devicetree path below to match your specific dtb file in the dtbs/ folder!
    devicetree /dtbs/qcom/sdm845-xiaomi-beryllium.dtb
}
GRUB_EOF

sudo chown -R $(whoami):$(whoami) esp_folder

echo ""
echo "======================================================="
echo "   ACTION REQUIRED: ESP FOLDER READY FOR MODIFICATION"
echo "======================================================="
echo "An unsealed ESP directory has been generated at: ./esp_folder/"
echo "Please verify your kernel files and edit ./esp_folder/EFI/BOOT/grub.cfg"
echo "to ensure the 'devicetree' path exactly matches the .dtb file."
echo ""
read -p "Press ENTER when you are done making changes to seal the ESP image..."

# --- Sealing the ESP ---
echo ">>> [ESP Seal] Generating 128MB FAT32 esp.img..."
dd if=/dev/zero of=esp.img bs=1M count=128 status=none
mkfs.vfat -F 32 esp.img
mkdir -p mnt_esp
sudo mount -o loop esp.img mnt_esp/
sudo cp -r esp_folder/* mnt_esp/
sudo umount mnt_esp
rm -rf mnt_esp

echo ">>> ESP successfully sealed into esp.img!"
echo ""
echo "======================================================="
echo "   Chroot Configuration Phase"
echo "======================================================="
echo "The base ubuntu_rootfs folder is fully populated and unsealed."
echo "Do you want to seal the RootFS into an image right now,"
echo "or continue making manual edits/configurations inside the chroot later?"
echo ""
echo "1) Seal it now (Generate final .img and flashing instructions)"
echo "2) Leave it unsealed (I will make edits and seal it later)"
read -p "Choice [1-2, default 2]: " CHROOT_CHOICE
CHROOT_CHOICE=${CHROOT_CHOICE:-2}

if [ "$CHROOT_CHOICE" == "1" ]; then
    echo ">>> Proceeding to seal..."
    bash 6_seal_rootfs.sh
else
    echo ">>> "
    echo ">>> Chroot left open at ./ubuntu_rootfs"
    echo ">>> To make edits later, run: bash 5_enter_chroot.sh"
    echo ">>> When you are finished, run: bash 6_seal_rootfs.sh"
    echo ">>> "
    echo ">>> Phase 4 complete."
fi
EOF_4

echo ">>> Entering chroot as root. Type 'exit' to leave."
echo "-------------------------------------------------------"

echo "-------------------------------------------------------"

echo ">>> Unmounting virtual filesystems..."
sudo umount ubuntu_rootfs/dev/pts || true
sudo umount ubuntu_rootfs/dev || true
sudo umount ubuntu_rootfs/proc || true
sudo umount ubuntu_rootfs/sys || true

echo ">>> Exited successfully. Run 6_seal_rootfs.sh when you are ready to pack the image."

cat << 'EOF_6' > 6_seal_rootfs.sh
#!/bin/bash
set -e
source build.env

echo "======================================================="
echo "   [6/6] Finalizing and Sealing RootFS"
echo "======================================================="
if [ ! -d "ubuntu_rootfs" ]; then
    echo ">>> [ERROR] ubuntu_rootfs directory not found."
    exit 1
fi

OUTPUT_IMG="ubuntu_beryllium.img"

# Ensure virtual filesystems aren't stuck mounted before packing
sudo umount ubuntu_rootfs/dev/pts 2>/dev/null || true
sudo umount ubuntu_rootfs/dev 2>/dev/null || true
sudo umount ubuntu_rootfs/proc 2>/dev/null || true
sudo umount ubuntu_rootfs/sys 2>/dev/null || true

IMG_MB=$((IMAGE_SIZE * 1024))
echo ">>> [Packing] Allocating ${IMAGE_SIZE}GB RootFS and packing..."
dd if=/dev/zero of="$OUTPUT_IMG" bs=1M count=$IMG_MB status=progress
mkfs.ext4 -L pmOS_root "$OUTPUT_IMG"
mkdir -p mnt_final
sudo mount -o loop "$OUTPUT_IMG" mnt_final/
sudo cp -a ubuntu_rootfs/. mnt_final/
sudo umount mnt_final
sudo e2fsck -f -y "$OUTPUT_IMG"

echo ">>> [Cleanup] RootFS sealed! The ubuntu_rootfs folder is no longer needed."
read -p "Do you want to delete the unsealed ubuntu_rootfs folder to save space? (y/n) [default: y]: " DEL_CHROOT
DEL_CHROOT=${DEL_CHROOT:-y}
if [ "$DEL_CHROOT" == "y" ]; then
    sudo rm -rf ubuntu_rootfs
    echo ">>> ubuntu_rootfs deleted."
fi

echo ""
echo "======================================================="
echo "   Flashing & Deployment Options"
echo "======================================================="
read -p "How do you plan to flash? (1: SD Card, 2: Fastboot/Internal) [Default: 1]: " FLASH_CHOICE
FLASH_CHOICE=${FLASH_CHOICE:-1}

if [ "$FLASH_CHOICE" == "2" ]; then
    echo ">>> [Fastboot Prep] Converting raw ext4/fat32 images to sparse images..."
    if ! command -v img2simg &> /dev/null; then
        sudo apt-get update && sudo apt-get install -y android-sdk-libsparse-utils || sudo apt-get install -y android-tools-fsutils
    fi
    
    img2simg "$OUTPUT_IMG" "${OUTPUT_IMG%.img}_sparse.img"
    if [ ! -f "esp_sparse.img" ] && [ -f "esp.img" ]; then
        img2simg esp.img esp_sparse.img
    fi
    
    echo ">>> "
    echo ">>> LOCATION:"
    echo ">>> Sealed ESP (Sparse): $(pwd)/esp_sparse.img"
    echo ">>> RootFS (Sparse):     $(pwd)/${OUTPUT_IMG%.img}_sparse.img"
    echo ">>> "
    echo ">>> FLASHING INSTRUCTIONS (Fastboot with UEFI):"
    echo ">>> 1. Reboot your Beryllium into Fastboot mode."
    echo ">>> 2. Flash the ESP sparse image (replace 'boot' with your target UEFI boot partition):"
    echo ">>>    fastboot flash boot $(pwd)/esp_sparse.img"
    echo ">>> "
    echo ">>> 3. Flash the RootFS sparse image (e.g., to userdata):"
    echo ">>>    fastboot flash userdata $(pwd)/${OUTPUT_IMG%.img}_sparse.img"
    echo ">>> "
    echo ">>> 4. Reboot your device: fastboot reboot"
else
    echo ">>> "
    echo ">>> LOCATION:"
    echo ">>> Sealed ESP Image: $(pwd)/esp.img"
    echo ">>> RootFS Image:     $(pwd)/$OUTPUT_IMG"
    echo ">>> "
    echo ">>> FLASHING INSTRUCTIONS (SD Card for UEFI):"
    echo ">>> Note: For UEFI booting from SD, you must manually create a GPT partition table"
    echo ">>> with a FAT32 ESP (Partition 1) and ext4 RootFS (Partition 2)."
    echo ">>> "
    echo ">>> 1. Flash the ESP to Partition 1:"
    echo ">>>    sudo dd if=$(pwd)/esp.img of=/dev/sdX1 bs=4M status=progress"
    echo ">>> "
    echo ">>> 2. Flash the RootFS to Partition 2:"
    echo ">>>    sudo dd if=$(pwd)/$OUTPUT_IMG of=/dev/sdX2 bs=4M status=progress"
fi
echo "======================================================="
EOF_6
